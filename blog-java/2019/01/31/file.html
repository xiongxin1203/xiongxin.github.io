前言

最近接到一个需求，要将一批多层次文件夹里面的文件根据其中某一层合并为一个个PDF文件，并将文件上传到固定的服务器上，要求处理的文件包括ppt，excel，word以及各种图片格式，其中图片格式要求压缩。
因为只是脚本，所以基于最基础的Springboot+gradle来构建框架。

应用到的主要技术有：

office文件转pdf
aspose(收费)

图片压缩
等宽压缩（实测显示效果比较好，但是压缩大小不是很理想。需求相反的话推荐用Thumbnails）

图片转pdf
itext

pdf合成
itext

图片格式转化
jai（部分图片格式用itext转pdf会报错，先转化为其他格式再转pdf）

读取excel文件
poi

上传文件
HttpCient

输出csv文件
csv

/**
 * word转PDF的license验证
 *
 * @return boolean
 */
private boolean wordMatchLicense() {
    boolean result = false;
    InputStream is = TransToPdfService.class.getClassLoader().getResourceAsStream("license/license.xml");
    com.aspose.words.License wordLicense = new com.aspose.words.License();
    try {
        wordLicense.setLicense(is);
        result = true;
    } catch (Exception e) {
        log.warn("载入word授权文件失败");
    }
    return result;
}

/**
 * word转PDF
 *
 * @param sourceFile File
 * @param targetFile File
 */
private void wordToPdf(File sourceFile, File targetFile) {
    FileOutputStream os = null;
    try {
        os = new FileOutputStream(targetFile);
        Document doc = new Document(sourceFile.getAbsolutePath());
        doc.save(os, com.aspose.words.SaveFormat.PDF);
    } catch (FileNotFoundException e) {
        log.error("输出到" + targetFile.getAbsolutePath() + "错误:" + e);
    } catch (Exception e) {
        log.error("转换word出错:" + e);
    } finally {
        if (os != null) {
            try {
                os.close();
            } catch (IOException e) {
                log.error("关闭输出流出错:" + e);
            }
        }
    }
}

/**
 * excel转PDF
 *
 * @param sourceFile File
 * @param targetFile File
 */
private void exlToPdf(File sourceFile, File targetFile) {
    FileOutputStream fileOs = null;
    Workbook wb;
    try {
        fileOs = new FileOutputStream(targetFile);
        wb = new Workbook(sourceFile.getAbsolutePath());
        WorksheetCollection worksheets = wb.getWorksheets();
        for (Object o : worksheets) {                 //遍历sheet，如果确定只需要某个sheet也可以不这么写
            Worksheet worksheet = (Worksheet) o;
            Cells cells = worksheet.getCells();
            int rowCount = cells.getMaxDataRow() + 1;   //存在数据的最大行数
            int columnCount = cells.getMaxDataColumn() + 1;   //存在数据的最大列数
            int allRowCount = cells.getMaxRow() + 1;   //最大行数
            int allColumnCount = cells.getMaxColumn() + 1;   //最大列数
            cells.unhideRows(0, rowCount, 18);         //取消隐藏
            cells.deleteRows(rowCount, (allRowCount - rowCount));        //行截取
            cells.deleteColumns(columnCount, (allColumnCount - columnCount), false);   //列截取
        }
        wb.save(fileOs, com.aspose.cells.SaveFormat.PDF);
    } catch (FileNotFoundException e) {
        log.error("输出到" + targetFile.getAbsolutePath() + "错误:" + e);
    } catch (Exception e) {
        log.error("转换excel出错:" + e);
    } finally {
        if (fileOs != null) {
            try {
                fileOs.close();
            } catch (IOException e) {
                log.error("关闭输出流出错:" + e);
            }
        }
    }
}

图片压缩

直接上代码:

/**
 * 等宽压缩
 *
 * @param w Float
 * @return Integer
 */
private Integer getPercent(Float w) {
    Integer p;
    Float p2;
    p2 = 530 / w * 100;
    p = Math.round(p2);
    return p;
}

你还可以选择等高压缩，等比例压缩等，这些百度都能查到，可以实际压缩看一下哪种效果比较满意，我最终选择了等宽压缩

图片转pdf

直接上代码:

/**
 * 图片转PDF文件
 *
 * @param sourceFile File
 * @param targetFile File
 */
private void picToPdf(File sourceFile, File targetFile) {
    com.itextpdf.text.Document doc = null;
    try {
        doc = new com.itextpdf.text.Document()
        PdfWriter.getInstance(doc, new FileOutputStream(targetFile.getAbsoluteFile()));
        doc.open();
        Image img = Image.getInstance(sourceFile.getAbsolutePath());
        Float width = img.getWidth();
        Integer percent = getPercent(width);    //调用压缩图片的方法
        img.setAlignment(Image.MIDDLE);
        img.scalePercent(percent);
        doc.add(img);
    } catch (Exception e) {
        log.error(e.getMessage());
    } finally {
        if (doc != null) {
            doc.close();
        }
    }
}

这里实例化了一个itext的Document和PdfWriter，PdfWriter负责将Document用流写入目标文件，后面都是针对Document加入图片的操作，其实也可以加入其他的东西，比如文字什么的

这里如果处理的都是图片，可以选择传入fileList，这样可以不用再进行pdf合并这个操作，我这边由于不只是处理图片，所以还是选择了单个转最后一起合并的方法

pdf合成

直接上代码:

/**
 * 合并PDF文件
 *
 * @param files   String[]
 * @param newFile String
 * @return boolean
 */
public boolean mergePdfFiles(String[] files, String newFile) {
    int index = 0;
    for (int i = 0; i < files.length; i++) {
        if (getFileFullName(files[i]).contains("申请")) {     //添加了一个小功能，根据文件名关键字把重要的文件放在前面
            String temp = files[index];
            files[index] = files[i];
            files[i] = temp;
            index++;
        }
    }
    log.info("开始合并" + newFile + ":_________________合并文件中");
    boolean retValue = false;
    com.itextpdf.text.Document document = null;
    try {
        document = new com.itextpdf.text.Document(new PdfReader(files[0]).getPageSize(1));
        PdfCopy copy = new PdfCopy(document, new FileOutputStream(newFile));
        document.open();
        for (int i = 0; i < files.length; i++) {
            PdfReader reader = new PdfReader(files[i]);
            int n = reader.getNumberOfPages();
            for (int j = 1; j <= n; j++) {
                document.newPage();
                PdfImportedPage page = copy.getImportedPage(reader, j);
                copy.addPage(page);
            }
        }
        retValue = true;
        log.info("合并完成" + newFile + ":_________________合并已完成");
    } catch (Exception e) {
        log.error(e.getMessage());
    } finally {
        if (document != null) {
            document.close();
        }
    }
    return retValue;
}

这里其实和图片转pdf差不多，只是工具变了。实例化了一个itext的Document和PdfCopy，PdfCopy负责将Document和PdfCopy的内容用流写入目标文件。首先遍历fileList，用PdfReader读取文件，获取文件页数，然后遍历页数，Document每次增加一页，然后用PdfCopy获取PdfReader中对应页的内容赋值给pdfImportedPage，再将pdfImportedPage加入到PdfCopy中

这里添加了一个小功能，可以根据某个关键字做一个置顶的操作，多写几个实现多关键字置顶也可以（getFileFullName是把文件path转为文件name，用最后一个”\”切割就可以了）

/**
 * 图片转为JPEG格式
 *
 * @param srcFilePath  String
 * @param descFilePath String
 */
private void jpgToJpeg(String srcFilePath, String descFilePath) {
    OutputStream os = null;
    try {
        RenderedOp src = JAI.create("fileload", srcFilePath);
        os = new FileOutputStream(descFilePath);
        JPEGEncodeParam param = new JPEGEncodeParam();
        ImageEncoder enc = ImageCodec.createImageEncoder("JPEG", os, param);
        enc.encode(src);
    } catch (Exception e) {
        log.error(e.getMessage());
    } finally {
        if (os != null) {
            try {
                os.close();
            } catch (IOException e) {
                log.error("关闭输出流出错:" + e);
            }
        }
    }
}

这里jai读取了图片的相关信息，有兴趣可以点开RenderedOp看下；然后实例化了ImageEncoder并指定了新文件类型，输出流以及转换参数（EncodeParam）；最后使用ImageEncoder转换源文件

EncodeParam：tif和tiff对应TIFFEncodeParam，jpg和jpeg对应JPEGEncodeParam，png对应PNGEncodeParam，bmp对应BMPEncodeParam，还有一些PNMEncodeParam之类的比较少见，感兴趣可以翻看源码

/**
 * 读取后缀为xlsx的excel文档
 *
 * @param path 文件路径
 * @return 合同号信息实体类
 */
private List<ContractInfo> xlsxFileRead(String path) {
    XSSFWorkbook excel = null;
    try {
        InputStream is = new FileInputStream(path);
        excel = new XSSFWorkbook(is);
    } catch (Exception e) {
        log.error(e.getMessage());
    }
    ContractInfo contractInfo;                      //创建容器
    List<ContractInfo> list = new ArrayList<>();    //创建容器list
    XSSFSheet sheet = excel.getSheetAt(0);          //这里直接获取第一个sheet，如果需要操作多个sheet需要全部获取并遍历
    for (int rowNum = 1; rowNum < sheet.getLastRowNum(); rowNum++) {
        XSSFRow row = sheet.getRow(rowNum);
        if (row == null) {
            continue;
        }
        contractInfo = new ContractInfo();
        XSSFCell cell0 = row.getCell(0);            //获取第一列cell
        if (cell0 == null) {
            continue;
        }
        contractInfo.setFileName(cell0.getStringCellValue());        //这里类型一定要对上，不然报错
        XSSFCell cell2 = row.getCell(1);            //获取第二列cell
        if (cell2 == null) {
            continue;
        }
        contractInfo.setContractName(cell2.getStringCellValue());       //这里类型一定要对上，不然报错
        list.add(contractInfo);
    }
    return list;
}

/**
 * 将回文写入csv文件留档
 *
 * @param headers 文件标题
 * @param data    文件数据
 */
private void writeCsv(Object[] headers, List<Object[]> data, String desFile) {
    CSVPrinter printer = null;
    Writer writer = null;
    try {
        File fileCsv = new File(desFile);
        if (!fileCsv.exists()) {
            fileCsv.createNewFile();
        }
        CSVFormat formatter = CSVFormat.DEFAULT.withRecordSeparator("\n");       //每行数据默认以换行分隔
        writer = new BufferedWriter(new BufferedWriter(new OutputStreamWriter(new FileOutputStream(fileCsv), "GBk")));  //不指定编码的话会出现中文乱码的现象
        printer = new CSVPrinter(writer, formatter);
        printer.printRecord(headers);
        if (null != data) {
            for (Object[] lineData : data) {
                printer.printRecord(lineData);
            }
        }
        writer.flush();
    } catch (Exception e) {
        log.error(e.getMessage());
    }  finally {
        try {
            if (printer != null) {
                printer.close(true);
            }
            if (writer != null) {
                writer.close();
            }
        } catch (IOException e) {
            log.error("关闭输出流出错:" + e);
        }
    }
}

使用的是apache commons内的csv包；将标题数据和内容数组列表传进来，先把标题写入csv文件，在遍历内容数据列表一行一行的把数据写入csv文件即可

需要指定编码，否则会中文乱码
